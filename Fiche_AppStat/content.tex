\chapter*{Notations}

\begin{itemize}
    \item $\mathcal{X}$ désigne l'ensemble des entrées (observations). $\mathcal{X}^m$ est l'ensemble des $m$-uplets $(x_1, \dots, x_m)$
    dont chaque terme est un élément de $\mathcal{X}$, on pose $\mathcal{X}^{*} = \bigcup_{m=1}^{+\infty} \mathcal{X}^{m}$.
    En d'autres tertmes, 
    \[
    \mathcal{X}^m = 
    \begin{bmatrix}
        x_1, \dots, x_m
    \end{bmatrix}
    \quad x_i \in \mathcal{X}, \forall i \in \{1;m\}
    \]
    \[
    \quad \mathcal{X}^{*} =\bigcup_{m=1}^{+\infty} \mathcal{X}^{m} =
    \begin{bmatrix}
        x_{1,1} & \dots & x_{1,m} \\
        \vdots   & \dots & \vdots \\
        x_{n,1} & \dots & x_{n,m}
    \end{bmatrix}
    \]
    \newline
    \item $|\mathcal{X}|$ désigne le cardinal de $\mathcal{X}$ \newline 
    \item $\mathcal{Y}$ désigne l'espace des sorties. On a aussi $\mathcal{Y}^m$ est l'ensemble des $m$-uplets $(y_1, \dots, y_m)$
    dont chaque terme est un élément de $\mathcal{Y}$, on pose $\mathcal{Y}^{*} = \bigcup_{m=1}^{+\infty} \mathcal{Y}^{m}$. 
    Il vient :
    \begin{equation}
        S = ((x_i, y_i)_{i \in \{1;m\}}) \in (\mathcal{X} \times \mathcal{Y})^{*} \nonumber
    \end{equation}
    \item $L$ désigne la fonction de coût \newline
    \item $m(\mathcal{X}, \mathcal{Y})$ est l'ensemble des fonctions mesurables \footnote{
        ($\mathcal{X}, m_{\mathcal{X}}$) et ($\mathcal{Y}, m_{\mathcal{Y}}$) sont des espaces mesurables
    } \newline
    \item $m_{\mathcal{H}}(\epsilon, \delta)$ est la fonction de complexité d'échantilon. \newline
    \item $(ERM)_S$ est l'hypothèse qui minimise le risque empirique sur $S$, i.e., 
    \begin{equation}
        (ERM)_{\mathcal{H}}(S) = A(S) = h_S = \text{arg} \min_{h \in \mathcal{H}} \mathcal{R}_S(h) \nonumber
    \end{equation}
    C'est la meilleure hypothèse pour le \textit{dataset} $S$. \newline 
    \item ($\Omega, \mathcal{E}, \mathbb{P}$) désigne un espace probabilisé \newline
    \item $\mathbf{A}$ désigne une matrice. \newline
    \item $\mathcal{M}_{n,p}(\mathbb{K})$ désigne l'espace des matrices de $n$ lignes, $p$ colonnes à coefficients dans $\mathbb{R}$
\end{itemize}

\chapter{Modèle formel de l'apprentissage statistique supervisé}
\section{Définitions générales}

\begin{itemize}

\item \noindent Soient $\mathcal{X}, \mathcal{Y}$ donnés. La fonction de coût $L$ est définie par : 

\[
\begin{aligned}
    L : \mathcal{Y} \times \mathcal{Y} &\to \mathbb{R}^{+} \\
    (y,y') &\mapsto L(y,y')
\end{aligned}
\]

\noindent C'est une fonction arbitraire. En général, on prend $L(y,y') = |y - y'|^2$ (méthode OLS) avec $y' = \hat{y} = h(x)$, 
$h$ l'hypothèse que l'on fait sur $x \in \mathcal{X}$ pour estimer la valeur $y$ i.e. $\hat{y}$ est un estimateur de $y$. $L$ est donc le coût
de décision de $h$ sur $x$. \newline

\item $h \in \mathcal{H} \subset m(\mathcal{X}, \mathcal{Y})$ avec $\mathcal{H}$ l'ensemble des hypothèses faites sur $x$ pour estimer $y$. \newline

\item $S = ((x,y_i))_{1 \leq i \leq m} = ((x_1,y_1), \dots, (x_m,y_m)) \in (\mathcal{X} \times \mathcal{Y})^{*}$  est une séquence d'objets labellisés.
Les $x_i$ sont les données observées et les $y_i$ les valeurs à estimées. $S$ représente le jeu de données d'entraînement \newline

\item Un Algorithme Automatique Supervisé (AAS) est une fonction tel que : 

\[
\begin{aligned}
A : (\mathcal{X} \times \mathcal{Y})^{*} &\to \mathcal{H} \\
S &\mapsto A(S) = h_s
\end{aligned}
\]

\item Risque réel :

\begin{align}
    \mathcal{R}(h) &= \mathbb{E}[L(y,y')] \\
                   &= \int_{\mathcal{X} \times \mathcal{Y}} L(y, h(x))p(x,y)dxdy \nonumber\\
                   &= \int_{\mathcal{X} \times \mathcal{Y}} L(y, h(x))p(y | x)p(x)dxdy \nonumber
\end{align}


\item Risque empirique :

\begin{equation}
    \mathcal{R}_s(h) = \hat{\mathcal{R}}(h) = \frac{1}{m} \sum_{i=1}^{m} L(h(x_i), y_i)
\end{equation}


\newpage

\noindent C'est le seul risque dont on a accès (quantité d'information disponible limitée et finie). On cherche à minimiser 
le risque réel $\mathcal{R}$ à l'aide de la décision prise parmis $\mathcal{H}, h_s = A(S)$, à savoir :

\begin{equation}
    \mathcal{R}(h_s) = \min_{h \in \mathcal{H}}\mathcal{R}(h) \Longleftrightarrow 
    (\text{ERM})_{\mathcal{H}}  = A(S) = h_s = \text{arg}\min_{h \in \mathcal{H}} \mathcal{R}_s(h)
\end{equation}

\noindent \textbf{Propriété :} $h \in \mathcal{H}, \mathcal{R}_s(h) \xrightarrow[m \to \infty]{p.s.} \mathcal{R}(h)$ et en moyenne quadratique. 
\footnote{Attention : $A(S)$ est une variable aléatoire donc $\mathcal{R}(A(S)) = \mathcal{R}(h_s)$ est une VA fonction de $S$, on ne peut pas appliquer la loi des grands nombres.}

\end{itemize}

\section{Capacité d'apprentissage PAC}
\subsection{Définition}

\noindent Soient $\mathcal{X}, \mathcal{Y}, L, \mathcal{H}$ fixés. $\mathcal{H}$ a la capacité PAC 
 \footnote{Probablement Approximativement Correct. Probablement car la probabilité de l'évènement est non négligeable ($\geq 1 - \delta$),
 Approximativement car le minimum global n'est pas atteint mais il s'en rapproche d'une distance au plus $\epsilon$ ($\geq \epsilon$)}
quand $\exists A, \exists m_{\mathcal{H}} : ]0;1[^{2} \to \mathbb{N}, \forall P, \forall (\epsilon, \delta) \in  ]0;1[^{2}, 
\forall m \ge m_{\mathcal{H}}(\epsilon, \delta), \forall S$

\begin{equation}
    P(\mathcal{R}(h_S) - \min_{h \in \mathcal{H}}\mathcal{R}(h) \leq \epsilon) \geq 1 - \delta \quad (*) \nonumber
\end{equation}

\noindent De même :

\begin{equation}
    m_{\mathcal{H}}(\epsilon, \delta) = \min \{m \in \mathbb{N} \quad \text{tel que} \quad \forall P, \forall S, \quad \text{(*) est vérifiée}\}
\end{equation}

\noindent $m_{\mathcal{H}}(\epsilon, \delta)$ est la taille minimale du \textit{dataset} qui permet à l'algorithme d'apprendre correctement. C'est
la fonction de complexité de $\mathcal{H}$ en mémoire ou en échantilon (données). Plus $\mathcal{H}$ est complexe, plus il faut de données. De même, plus $\epsilon, \delta$ sont petits,
plus il faut de données. Si $\epsilon$ et $\delta$ sont petits, cela veut dire que l'algorithme retourne un bon modèle avec une grande probabilité. En général, pour une classe de VC-dimension $d$, on prend 
$m_{\mathcal{H}}(\epsilon, \delta) \approx \frac{d + log(\frac{1}{\delta})}{\epsilon}$\newline

\noindent \textbf{Propriété :} Soit $\epsilon > 0$, $S$ est \textbf{$\epsilon$-représentatif} si et seulement si pour $h_s = (ERM)_{\mathcal{H}}(S)$,

\begin{equation}
    \mathcal{R}(h_s) - \min_{h \in \mathcal{H}} \mathcal{R}(h) \leq \epsilon
\end{equation}

\newpage

\subsection{Convergence uniforme}

\noindent $\mathcal{H}$ a la propriété de convergence uniforme quand $\exists m_{\mathcal{H}}^{CU} \in ]0;1[^{2} \rightarrow \mathbb{N},
\forall P, \forall (\epsilon, \delta) \in  ]0;1[^{2}, \forall m \geq m_{\mathcal{H}}^{CU}(\epsilon, \delta), \forall S,$

\begin{equation}
    P(\text{S est espilon-représentatif}) \geq 1 - \delta \Longleftrightarrow
    P(\sup_{h \in \mathcal{H}} | \mathcal{R}(h) - \mathcal{R}_S(h) | \leq \epsilon) \geq 1 - \delta \quad (**) \nonumber
\end{equation}

\noindent \textbf{Propriété :} Si $\mathcal{H}$ a la propriété de CU avec $m_{H_{CU}}^{CU}$ alors $\mathcal{H}$ a la capacité PAC par $A = (ERM)_{\mathcal{H}}$ et 

\begin{equation}
    m_{H}(\epsilon, \delta) \leq m_{H}^{CU}(\frac{\epsilon}{2}, \delta)
\end{equation}

\subsection{Capacité d'apprentissage PAC pour des ensemble finis ($|\mathcal{H}| < \infty$)}

\noindent \textbf{Lemme d'Hoeffding (1962)} : pour $(X_i)_{1 \geq i \geq m}$ VA i.i.d., $\mathbb{E}[X_i] = \mu, \exists a < b, a,b \in, \mathbb{R}$
tel que $P(X_i \in [a;b]) = 1$, alors, $\forall \epsilon > 0$,

\begin{equation} 
    P(| \frac{1}{m}\sum_{i=1}^{m}X_i - \mu|) \leq 2e^{ \frac{-2m\epsilon^2}{(b-a)^2}}
\end{equation}

\noindent On appelle cette inégalité inégalité de concentration. D'après Hoeffding, on a que $\mathcal{H}$ a la propriété de CU SSI:

\begin{equation}
    m_{\mathcal{H}}^{CU}(\epsilon, \delta) \leq 
    \left\lceil \frac{\log\left(\frac{2|\mathcal{H}|}{\delta}\right)}{2\epsilon^2} \right\rceil
\end{equation}

\noindent et donc $\mathcal{H}$ a la capacité PAC avec 

\begin{equation}
    m_{\mathcal{H}}(\epsilon, \delta) \leq 
    m_{\mathcal{H}}^{CU}(\frac{\epsilon}{2}, \delta) \leq
    \left\lceil \frac{2\log(\frac{2|\mathcal{H}|}{\delta})}{\epsilon^2} \right\rceil
\end{equation}

\subsection{Capacité d'apprentissage PAC faible}

\noindent $\mathcal{H}$ a la capacité d'apprentissage PAC faible quand $\exists A$ AAS, $\exists m_{\mathcal{H}} : ]0;1[^2 \leftarrow \mathbb{N},
\forall (\epsilon, \delta) \in ]0;1[^2, \forall  m \geq m_{\mathcal{H}}(\epsilon, \delta), \forall P, \exists h \in \mathcal{H}$,

\begin{itemize}

    \item \begin{equation}
        \mathcal{R}(h)= 0 
    \end{equation}

    \item \begin{equation}
        P(\mathcal{R}(h_S) \geq \epsilon) \leq \delta 
    \end{equation}

\end{itemize}

\noindent Naturellement, $\mathcal{H}$ n'as pas la CA PAC si $\mathcal{H}$ n'a pas la CA PAC faible

\section{Dilemme biais / complexité (\textit{underfitting vs. overfitting})}
\subsection{Définition}

\noindent Pour certains algorithmes comme les $K$ plus proches voisins, la complexité devient énorme, ce qui nous pousse à trouver un équilibre entre la 
quantité de données disponibles et la performance du modèle.

\begin{equation}
    \mathcal{R}(h_S) = \underbrace{\mathcal{R}(h_S) - \min_{h \in \mathcal{H}} \mathcal{R}(h)}_{\epsilon \text{ d'estimation}} +  \underbrace{\min_{h \in \mathcal{H}} \mathcal{R}(h)}_{\epsilon \text{ approx.}}
\end{equation}

\[
\phantom{\mathcal{R}(h_S) =}
\quad \downarrow \hspace{3cm} \downarrow
\]

\[
\phantom{\mathcal{R}(h_S) =}
\quad \text{variance} \hspace{2cm} \text{biais}
\]

\begin{itemize}
    \item Si $|\mathcal{H}|$ est trop petit, l'espace des hypothèses n'est pas assez complexe pour trouver une hypothèse qui apprend bien les données (\textit{underfitting}). 
    Dans ce cas, le \textbf{biais} $\searrow$ \textbf{et la variance} $\nearrow$ \newline
    \item Si $|\mathcal{H}|$ est trop grand, l'espace des hypothèses est très complexe et peut coller parfaitement aux données ($\mathcal{R}_S(h) \rightarrow 0$) mais généralise mal (\textit{overfitting}). 
    Il a aussi besoin de plus de données. Dans ce cas, le \textbf{biais} $\nearrow$ \textbf{et la variance} $\searrow$  
\end{itemize}

\subsection{Il n'existe pas d'aprocimateur universel}

\noindent \textbf{Lemme 1 :} Soit $m \in \mathbb{N}^{*}$, si $|\mathcal{X}| \leq 2m$ et $\mathcal{H} \in \mathcal{M}(\mathcal{X}, \mathcal{Y})$,
alors $\forall A \in \mathcal{H}$ AAS, $\exists \mathbb{P}$ sur $\mathcal{X} \times \mathcal{Y}$ tel que : 

\begin{equation}
    \exists f \in \mathcal{H}, \quad \mathcal{R}(f) = 0 \nonumber
\end{equation}

\begin{equation}
    \forall S \in (\mathcal{X} \times \mathcal{Y})^* i.i.d., \quad \mathbb{E}[\mathcal{R}(S)] \geq \frac{1}{4} \nonumber
\end{equation}

\noindent \textbf{No free lunch theorem : } Avec les hypothèses du lemme précédent et $\forall A \in \mathcal{H}$ AAS, $\exists \mathbb{P}$ sur $\mathcal{X} \times \mathcal{Y}$

\begin{equation}
    \exists f \in \mathcal{H}, \quad \mathcal{R}(f) = 0 \nonumber
\end{equation}

\begin{equation}
    \forall S \in (\mathcal{X} \times \mathcal{Y})^* i.i.d., \quad \mathbb{P}(\mathcal{R}[A(S)] \geq \frac{1}{8}) \geq \frac{1}{7} \nonumber
\end{equation}

\noindent Le \textit{no free lunch theorem} explique que même si il existe une hypothèse de telle sorte que le problème soit parfaitement aprenable, il est fort probable que
le modèle appris ait un risque non négligeable. Sans hypothèses sur $\mathbb{P}$, l'apprentissage est peu fiable.

\section{Dimension VC (Vapnik-Chervonenkis)}
\subsection{Définitions}

\begin{itemize}

    \item On définit le \textbf{détecteur à seuil} comme étant la fonction $h_a$ tel que :

    \[
    \begin{aligned}
        h_a : \mathbb{R} &\to \{ 0;1 \} \\
        x &\mapsto h_a(x) = \mathbf{1}_{[a,+\infty)}(x) =
                    \begin{cases}
                    1 & \text{si } x \geq a \\
                    0 & \text{si } x < a
                    \end{cases}
    \end{aligned}
    \]

    \begin{tikzpicture}
        \draw[->] (-2,0) -- (3,0);
        \draw[->] (0,-0.5) -- (0,1.5);
        \draw[thick] (-2,0) -- (1,0);
        \draw[thick] (1,1) -- (3,1);
        \filldraw[fill=white] (1,0) circle (2pt);
        \fill (1,1) circle (2pt);

        \node[below] at (1,0) {$a$};
    \end{tikzpicture}

    \item Soit $\mathcal{C} \subset \mathcal{X}$, la restriction de $\mathcal{H}$ à $\mathcal{C}$, notée $\mathcal{H}_{\mathcal{C}}$ tel que 

    \begin{equation}
        \mathcal{H}_{\mathcal{C}} = \{ h/c : \mathcal{C} \rightarrow \mathcal{Y}, h \in \mathcal{H} \}
    \end{equation}

    \noindent Quand $\mathcal{C} = (c_1, \dots, c_m), m < +\infty$, on idnetifie la restriction $h/c$ d'une hypothèse $h$ au $m$-uplet $(h(c_1), \dots, h(c_m)) \in \mathcal{Y}^{m}$.
    Autrement dit, \textbf{on regarde le comportement des hypothèses $h \in \mathcal{H}$ sur les points de $\mathcal{C} \subset \mathcal{X}$ uniquement} \newline

    \item Un ensemble d'hypothèses $\mathcal{H} \subset \mathcal{M}(\mathcal{X}, \{0;1\})$ brise $\mathcal{C}$ quand $\mathcal{H}_{\mathcal{C}}$ est l'ensemble 
    des fonctions de $\mathcal{C}$ dans $\{0;1\}$. i.e.

    \begin{equation}
        |\mathcal{H}_{\mathcal{C}}| = 2^{|\mathcal{C}|}
    \end{equation}

    \noindent Autrement dit, $\mathcal{H}$ \textbf{brise} $\mathcal{C}$ veut dire qu'il existe dans $\mathcal{H}$ une fonction $h$ qui est parfaitement capable de fitter les labels
    de $\mathcal{Y}$ sur $\mathcal{C}$. Il vient que $\mathcal{R}_S(h) = 0$ et que pour $\mathcal{C} = (c_1, \dots, c_m), \forall y = (y_1, \dots, y_m) \in \mathcal{Y}^{m} = \{0;1\}^{m},
    \exists h \in \mathcal{H}$ tel que 

    \begin{equation}
        y_i = h(x_i), \forall i \in \{1;m\}
    \end{equation}

\end{itemize}

\noindent la dimension VC d'un ensemble d'hypothèses $\mathcal{H}$, notée $\mathrm{dimVC}(\mathcal{H})$ est la taille maximale d'un sous-ensemble
$\mathcal{C} \subset \mathcal{X}$ qui peut être brisé par $\mathcal{H}$.

\begin{equation}
    \mathrm{dimVC}(\mathcal{H}) = \sup \{ |\mathcal{C}| \; : \; \mathcal{C} \subset \mathcal{X} \text{ est brisé par } \mathcal{H} \}.
\end{equation}

\noindent Si $\mathcal{H}$ est capable de briser un espace $\mathcal{C}$ de n'importe quelle taille, $\mathrm{dimVC}(\mathcal{H}) = +\infty$ \footnote{
    on écrit sup au lieu de max dans la définition car en effet $\mathrm{VCdim}$ peut être infinie.
}

\noindent \textbf{Théorème : } Aucun ensemble d'hypothèses $\mathcal{H}$ de dimension VC infinie n'a la CA PAC faible ou CA PAC.
